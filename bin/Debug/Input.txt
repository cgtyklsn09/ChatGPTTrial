using System;
using Inventor;

class Program
{
    public static void Main()
    {
        try
        {
            // Get the active Inventor application
            Inventor.Application m_inventorApp = System.Runtime.InteropServices.Marshal.GetActiveObject("Inventor.Application") as Inventor.Application;

            // Check if there is an active document
            if (m_inventorApp.ActiveDocument != null)
            {
                // Get the active document name
                string activeDocumentName = m_inventorApp.ActiveDocument.DisplayName;

                // Print the active document name
                Console.WriteLine("Active Document Name: " + activeDocumentName);
            }
            else
            {
                Console.WriteLine("No active document found.");
            }
        }
        catch (Exception ex)
        {
            Console.WriteLine("Error: " + ex.Message);
        }

        Console.ReadLine();
    }
}
